import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

serve(async (req) => {
  // CORS Preflight request
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  // 1. Log the incoming request to help debug
  console.log(`[Webhook Recebido] Method: ${req.method} | URL: ${req.url}`);

  try {
    // Parse query params and body safely
    const url = new URL(req.url);
    const queryId = url.searchParams.get("data.id") || url.searchParams.get("id");
    const queryType = url.searchParams.get("type") || url.searchParams.get("topic");

    let bodyText = "";
    let bodyData: any = {};
    try {
      bodyText = await req.clone().text();
      if (bodyText) {
        bodyData = JSON.parse(bodyText);
      }
    } catch (e) {
      console.log("No JSON body or unable to parse body:", e.message);
    }

    console.log("Headers:", JSON.stringify(Object.fromEntries(req.headers.entries())));
    console.log("Query Params:", JSON.stringify(Object.fromEntries(url.searchParams.entries())));
    console.log("Body:", bodyText);

    // Identify if this is a webhook/IPN from Mercado Pago
    // Mercado Pago typically sends webhooks with payment ID in query parameters or in body (e.g. bodyData.data?.id)
    // Client calls always send a specific client action ("create_pix", "create_preference", "check_status")
    const clientAction = bodyData.action;
    const isClientAction = ["create_pix", "create_preference", "check_status"].includes(clientAction);

    if (isClientAction) {
      // HANDLE CLIENT ACTIONS (create_pix, create_preference, check_status)
      const { token, transaction_amount, description, payer, items, payment_id, notification_url, external_reference } = bodyData;

      if (!token) {
        return new Response(JSON.stringify({ error: "Token ausente" }), {
          status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      if (clientAction === "create_pix") {
        const response = await fetch("https://api.mercadopago.com/v1/payments", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${token}`,
            "X-Idempotency-Key": `enki-${Date.now()}-${Math.floor(Math.random() * 1000)}`
          },
          body: JSON.stringify({
            transaction_amount,
            description,
            payment_method_id: "pix",
            payer,
            notification_url,
            external_reference
          })
        });

        const data = await response.json();
        if (!response.ok) {
          return new Response(JSON.stringify({ error: data.message || "Erro MP" }), {
            status: response.status, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
          });
        }
        return new Response(JSON.stringify({ paymentId: data.id, pixKey: data.point_of_interaction?.transaction_data?.qr_code }), {
          status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      if (clientAction === "create_preference") {
        const response = await fetch("https://api.mercadopago.com/checkout/preferences", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${token}`
          },
          body: JSON.stringify({ items, payer, notification_url, external_reference })
        });

        const data = await response.json();
        if (!response.ok) {
          return new Response(JSON.stringify({ error: data.message || "Erro MP" }), {
            status: response.status, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
          });
        }
        return new Response(JSON.stringify({ id: data.id, init_point: data.init_point }), {
          status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      if (clientAction === "check_status") {
        const response = await fetch(`https://api.mercadopago.com/v1/payments/${payment_id}`, {
          headers: { "Authorization": `Bearer ${token}` }
        });
        const data = await response.json();
        return new Response(JSON.stringify({ status: data.status }), {
          status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }
    } else {
      // HANDLE INCOMING WEBHOOK FROM MERCADO PAGO
      const paymentId = bodyData.data?.id || bodyData.id || queryId;
      const type = bodyData.type || bodyData.action || queryType;

      if (!paymentId) {
        console.log("Ignorando webhook: nenhum paymentId encontrado no payload ou query.");
        return new Response(JSON.stringify({ status: "ignored", reason: "no paymentId" }), {
          status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      // Initialize Supabase Client to fetch the Mercado Pago Access Token and update the orders
      const supabaseUrl = Deno.env.get("SUPABASE_URL") || "";
      const supabaseKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") || Deno.env.get("SUPABASE_ANON_KEY") || "";
      
      if (!supabaseUrl || !supabaseKey) {
        console.error("Erro no Webhook: Variáveis do Supabase ausentes no ambiente.");
        return new Response(JSON.stringify({ error: "Supabase config missing" }), {
          status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      if (Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")) {
        console.log("[Supabase Auth] Utilizando SUPABASE_SERVICE_ROLE_KEY para ignorar políticas RLS (Admin Privileges).");
      } else {
        console.warn("[Supabase Auth] SUPABASE_SERVICE_ROLE_KEY não encontrada! Utilizando fallback da chave anônima.");
      }

      const supabase = createClient(supabaseUrl, supabaseKey);

      // Fetch mp_access_token from loja_config
      const { data: configData, error: configError } = await supabase
        .from("loja_config")
        .select("mp_access_token")
        .eq("id", 1)
        .maybeSingle();

      if (configError) {
        console.error("Erro ao buscar loja_config no Supabase:", configError);
        return new Response(JSON.stringify({ error: "Database error" }), {
          status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      const mpAccessToken = configData?.mp_access_token?.trim();
      if (!mpAccessToken) {
        console.error("Erro no Webhook: mp_access_token está vazio na loja_config.");
        return new Response(JSON.stringify({ error: "Access token missing in database" }), {
          status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
        });
      }

      // Fetch payment status directly from Mercado Pago API using the token
      console.log(`Buscando detalhes do pagamento ${paymentId} no Mercado Pago...`);
      const mpResponse = await fetch(`https://api.mercadopago.com/v1/payments/${paymentId}`, {
        headers: {
          "Authorization": `Bearer ${mpAccessToken}`
        }
      });

      if (!mpResponse.ok) {
        const errorText = await mpResponse.text();
        console.error(`Erro ao buscar detalhes do pagamento no MP: ${mpResponse.status} - ${errorText}`);
        return new Response(JSON.stringify({ error: "MP API error" }), {
          status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } // MP expects 2xx even on failures so it won't retry excessively
        });
      }

      const payment = await mpResponse.json();
      const externalReference = payment.external_reference;
      const status = payment.status;

      console.log(`[Webhook MP] Status do pagamento ${paymentId}: ${status}. External Reference: ${externalReference}`);

      let updated = false;

      if (status === "approved" || status === "authorized") {
        console.log(`[Webhook UPDATE] Iniciando atualizações de status para 'Aprovado' no banco de dados.`);

        // Helper function to update status and/or gateway_status on a given column/value pair
        const performUpdate = async (matchColumn: string, matchValue: any) => {
          if (!matchValue) return false;
          console.log(`[Webhook DB UPDATE] Tentando atualizar registro onde ${matchColumn} = '${matchValue}'...`);

          try {
            // Attempt A: Update both status and gateway_status to 'Aprovado'
            console.log(`[Webhook DB UPDATE] Chamando .update({ status: 'Aprovado', gateway_status: 'Aprovado' }).eq('${matchColumn}', '${matchValue}')`);
            const { data, error } = await supabase
              .from("clientes_pedidos")
              .update({ 
                status: "Aprovado", 
                gateway_status: "Aprovado" 
              })
              .eq(matchColumn, matchValue)
              .select();

            if (!error && data && data.length > 0) {
              console.log(`[Webhook DB UPDATE SUCESSO] ${data.length} registro(s) atualizados com sucesso via ${matchColumn} = '${matchValue}' para status = 'Aprovado' e gateway_status = 'Aprovado'.`);
              return true;
            }

            if (error) {
              console.warn(`[Webhook DB UPDATE AVISO] Falha ao atualizar ambas as colunas via ${matchColumn}: ${error.message}. Tentando apenas gateway_status.`);
              
              // Attempt B (Fallback): Update only gateway_status
              const fallbackRes = await supabase
                .from("clientes_pedidos")
                .update({ gateway_status: "Aprovado" })
                .eq(matchColumn, matchValue)
                .select();

              if (!fallbackRes.error && fallbackRes.data && fallbackRes.data.length > 0) {
                console.log(`[Webhook DB UPDATE SUCESSO - FALLBACK] ${fallbackRes.data.length} registro(s) atualizados via ${matchColumn} = '${matchValue}' para gateway_status = 'Aprovado'.`);
                return true;
              }
            }
          } catch (e: any) {
            console.error(`[Webhook DB UPDATE EXCEÇÃO] Ocorreu uma exceção ao atualizar via ${matchColumn} = '${matchValue}':`, e.message);
          }
          return false;
        };

        // Strategy 1: Match gateway_id with externalReference (PED-...)
        if (externalReference) {
          updated = await performUpdate("gateway_id", externalReference);
        }

        // Strategy 2: Match id with externalReference
        if (!updated && externalReference) {
          updated = await performUpdate("id", externalReference);
        }

        // Strategy 3: Match gateway_id with paymentId from Mercado Pago
        if (!updated && paymentId) {
          updated = await performUpdate("gateway_id", paymentId);
        }

        // Strategy 4: Match id with paymentId from Mercado Pago
        if (!updated && paymentId) {
          updated = await performUpdate("id", paymentId);
        }

        if (updated) {
          console.log(`[Webhook Sucesso] Pedido associado a externalReference '${externalReference}' / paymentId '${paymentId}' foi devidamente Aprovado.`);
        } else {
          console.error(`[Webhook Falha] Nenhum pedido pôde ser localizado ou atualizado no banco de dados para os identificadores fornecidos.`);
        }
      } else {
        console.log(`[Webhook] Nenhuma ação de aprovação tomada. O status do pagamento no Mercado Pago é: ${status}`);
      }

      return new Response(JSON.stringify({ success: updated, paymentId, status, externalReference }), {
        status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      });
    }

    return new Response(JSON.stringify({ error: "Invalid request format" }), {
      status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
    });

  } catch (error) {
    console.error("Erro inesperado no Deno Edge Function:", error.message);
    return new Response(JSON.stringify({ error: error.message }), {
      status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
    });
  }
});
